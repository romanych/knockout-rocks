var
  circle = doc.byId('circle'),
  radiusField = doc.byId('radius'),
  colorField = doc.byId('color');


radiusField.onchange = function () {
  var radius = radiusField.value;
  circle.setAttribute('r', radius);
}

colorField.onchange = function () {
  var c = colorField.value;
  circle.setAttribute('stroke', c);
}

<svg>
  <circle
    id="circle"
    cx="105"
    cy="105"
    r="50"
    stroke="red"
    fill="transparent" />
</svg>
  
Radius:
<input type="range"
  min="10" max="100"
  step="1" id="radius"
  value="50"/>

Color:
<input type="text"
  id="color" value="red"/>
